install new-lg4ff
==================
; check your installed kernel version
apt search linux-headers-$(uname -r)

; check if kernel headers are already installed
ls -l /usr/src/linux-headers-$(uname -r)

; if not installed
sudo apt install linux-headers-$(uname -r)

; install dkms
sudo apt-get install dkms

;uninstall and remove
dkms uninstall new-lg4ff/-0.3.1 -k $(uname -r)
dkms remove new-lg4ff -v 0.3.1 --all
rm -r /var/lib/dkms/new-lg4ff
rm -r /usr/src/new-lg4ff
rm -r /usr/src/new-lg4ff-0.3.1
rm /usr/src/master.zip
rm /usr/src/fix32bits.zip
echo create folder
cd /usr/src/

;download package
wget https://github.com/berarma/new-lg4ff/archive/refs/heads/master.zip
or
wget https://github.com/berarma/new-lg4ff/archive/refs/heads/fix32bits.zip

; extract and copy to destination folder
sudo unzip ./master.zip
mv /usr/src/new-lg4ff-master /usr/src/new-lg4ff
or
sudo unzip ./fix32bits.zip
mv /usr/src/new-lg4ff-fix32bits /usr/src/new-lg4ff

;compile and install package
sudo dkms install /usr/src/new-lg4ff
